Convert the Frida20190612da_utf8.csv file into json

import csv
import json

with open('Frida20190612da_utf8.csv', 'r') as f:
    reader = csv.DictReader(f, delimiter=";")
    with open('frida.json', 'w') as j:
        j.write(json.dumps(list(reader)))
